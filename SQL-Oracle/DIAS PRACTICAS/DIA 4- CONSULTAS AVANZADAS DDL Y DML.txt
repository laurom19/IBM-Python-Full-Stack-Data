EN LAS CONSULTAS AVANZADAS EN VES DE FILTRAS TODOS LOS DATOS DE LAS TABLAS PODEMOS YA ESPECIFICAR QUE COLUMNAS/ATRIBUTOS DESEAMOS MOSTRAR. 
PARA HACERLA MAS AVANZADA Y FILTRAR MEJOR LOS DATOS A BUSCAR AÑADIR LA CONSULTA WHERE, ESTA NOS VA A DEVOLVER V/F O DESCONOCIDO, SIEMPRE Y EN CUANDO LE PONGAMOS EL PREDICADO O CONDICION.
EN EL CASO DE NO ACORDARNOS TODO EL DATO TENEMOS LA OPCION DE FILTRAR POR LIKE
UN EJEMPLO SERIA
SELECT ID_LIBROS, TITULO, AUTOR
FROM LIBROS
WHERE AUTOR LIKE 'E%'
EN EL LIKE EL % NOS SIRVE COMO UN COMODIN EN EL CUAL NOS PERMITE BUSCAR SIN RECORDAR TODO EL NOMBRE COMPLETO, Y EL % SE USA PARA INDICAR QUE NOS FALTAN LETRAS.
EJ: LOS AUTORES QUE COMIENCEN CON LA PRIMER LETRA CON 'E%' NOS TRAERIA DATOS COMO EDUARDO, EDGARDO, EDINSON,ETC QUE ESTEN EN LA BASE DE DATOS.

BUSQUEDAS DE RANGO
PARA BUSCAR LIBROS QUE TENGAN CIERTA CANTIDAD DE PAGINAS LO PODRIAMOS HACER DE LA SIGUIENTE MANERA

SELECT TITULO, PAGINAS
FROM LIBROS
WHERE PAGINAS >= 200 AND PAGINAS <= 300
ESTO NOS BUSCARIA PAGINAS MAYOR E IGUAL 200, Y MENOR E IGUAL 300.
TAMBIEN LO PODEMOS REALIZAR CON LA SENTENCIA BETWEEN 
SELECT TITULO, PAGINAS
FROM LIBROS
WHERE PAGINAS BETWEEN 290 AND 300

OTRA FORMA DE TRAER CONSULTAS MEDIANTE UNA CONDICION OR QUE ES SI ENCUENTRA UNO O OTRO
EJEMPLO NECESITAMOS SELECCIONAR AUTORES DE AUSTRALIA O BRASIL LO PODEMOS EJECUTAR DE LA SIGUIENTE MANERA

SELECT NOMBRE, APELLIDO, PAIS
FROM AUTOR
WHERE PAIS='AU' OR PAIS='BR'



ORDER BY
PARA REALIZAR CONSULTAS DE UN A COLUMNA/ATRIBUTO CON UN ORDEN ESPECIFICO USAMOS LA CLAUSULA ORDER BY, LA MISMA LA PODEMOS RELIZAR EN FORMA ASCENDENTE O DESCENTENTE, INDICANDOLE DESC O ASC, POR DEFECTO SIEMPRE ESTA EN FORMA ASCENDENTE.
OTRA FORMA DE ORDENACION ES INDICAR LA SECUENCIA DE LA COLUMNA
POR EJEMPLO 
SELECT TITULO, PAGINA
FROM LIBROS
ORDER BY 2.
ESTO INDICARIA QUE NOS ORDENARIA POR LAS PAGINAS YA QUE LE INDICAMOS EL INDICE 2 Y LO ORDENA EN FORMA ASCENDENTE.

EN EL CASO QUE TENEMOS UNA TABLA CON PAISES DONDE NACIERON LOS AUTORES, SE REPETIRIA MUCHOS PAISES, EN EL CASO DE QUERER TRAER LOS PAISES QUE NO SON DUPLICADOS UTILIZARIAMOS LA PALABRA CLAVE DISTINCT, ESO SOLO NOS TRAERIA LOS RESULTADOS NO DUPLICADOS.

SELECT DISTINCT (PAISES)
FROM AUTOR

EN EL CASO DE QUERER CONTABILIZAR CUANTOS SON DE EL MISMO PAIS UTILIZARIAMOS EN LA MISMA CONSULTA UN COUNT PARA HACER UN RECUENTO DE LOS MISMOS Y LUEGO PODEMOS HACER UN AGRUPAMIENTO CON LA CLAUSULA GROUP BY, ESTA NOS AGRUPA TAMBIEN LOS PAISES

SELECT PAISES, COUNT(PAISES)
FROM AUTOR
GROUP BY PAISES

EN LUGAR DE PONER COLUMNA 2, NOS PONE SOLAMENTE EL NUMERO, PARA RENOMBRAR LA COLUMNA PODEMOS EJECUTAR EL ALIAS, EN ESTA LE VAMOS A CAMBIAR EL NOMBRE QUE QUEREMOS

SELECT PAISES, COUNT (PAISES) AS 'SUMA PAIS'
FROM AUTOR 
GROUP BY PAISES

OTRO EJEMPLO PODEMOS COMPROBAR SI HAY MAS DE 4 AUTORES DEL MISMO PAIS UTILIZANDO LA PALABRA HAVING
SELECT PAISES, COUNT (PAISES) AS SUMA
FROM AUTOR
GROUP BY PAISES
HAVING COUNT (PAISES)-- > 4
